Python 3.0

 
import json
from urllib.request import urlopen url = "https://civicinsight.azurewebsites.net/athena"
response = urlopen(url)
contents = response.read()
text = contents.decode('utf8')
data = json.loads(text)
print(json.dumps(data, indent=4))
